

import React from 'react';
import '../../App.css';
import { useState, useEffect } from 'react';
import axios from 'axios';
import { Carousel } from 'react-responsive-carousel';
import "react-responsive-carousel/lib/styles/carousel.min.css"; 
import bgImage from '../../assets/images/poster.png';
import imdbLogo from '../../assets/images/imbd.png';
import orangeLogo from '../../assets/images/orange.png';
import playBtn from '../../assets/images/play.png';

function Hero() {
  const [movies, setMovies] = useState([]);

  useEffect(() => {
    const fetchMovies = async () => {
      try {
        const response = await axios.get(
          'https://api.themoviedb.org/3/discover/movie',
          {
            params: {
              api_key: '7f23fe1838341209abe18781249c152f',
              language: 'en-US',
              sort_by: 'popularity.desc',
              page: 1,
            },
          }
        );
        console.log(response.data.results);
        setMovies(response.data.results);
      } catch (error) {
        console.error('Error fetching data:', error);
      }
    };

    fetchMovies();
  }, []);


  return (
    <div>
      <Carousel       
      showThumbs={false}
        showStatus={false}
        showIndicators={false} 
        showArrows={false}
        infiniteLoop={true}
        autoPlay={true}
        interval={3000}
        > 


        {movies.map((movie) => (
          <div key={movie.id} className="hero-section w-full h-[100vh]  -z-10  bg-no-repeat bg-cover bg-center" style={{ backgroundImage: `url(${bgImage})` }}>
      <div className="movieDetail">
        <h1 className="movieTitle text-white font-bold">{movie.title}</h1>
        <div className="rating flex">
          <div className="rating1 flex gap-x-2">
            <img className="imdbLogo h-2" src={imdbLogo} alt="imdb logo" />
            <p className="rate text-white">{movie.vote_count}</p>
          </div>
          <div className="rating2 flex gap-x-2">
            <img className="orangeLogo" src={orangeLogo} alt="orange logo" />
            <p className="rate text-white">{movie.vote_average}%</p>
          </div>
        </div>
        <div className="movieDescription text-white">
          <p className=" text-ellipsis whitespace-nowrap">
            {movie.overview}
            {/* John Wick is on the run after killing a member <br/>of the international assassins'
          guild, and with a <br/> $14 million price tag on his head, he is the target <br/> of hit
          men and women everywhere. */}
          </p>
        </div>
        <button className='watchTrailerBtn text-white md:text-xl flex p-4 gap-x-3 items-center my-5 bg-custom-dark-Red-Color rounded-lg'> 
        <img className="watchTrailer w-5 h-5" src={playBtn} alt='watch trailer' />
            WATCH TRAILER
            </button>
      </div>
    </div>
        ))}
    </Carousel>
    </div>
  );
}

export default Hero;
